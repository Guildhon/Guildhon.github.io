```
一个星期有7天，按照下面格式输出。
输入：1 3 4 5 7
//全是升序输入，最大长度为7
输出：1，3-5，7
//三个以上（包括三个）的连续的用-输出

let arr=[1,3,4,5,7,8,9,12,13,55,56,57,58,80];
let putArr = [];
for (var i = 0; i < arr.length; i++){
	var flag = 1;
	var j = i;
	while (j < arr.length - 2){
		if (arr[j] + 1 == arr[j + 1]){
			flag++;
			j++;
		} else {
			break;
		}
	}

	if (flag >= 3) {
		putArr.push(arr[i]+"-"+arr[j]);
		i = j;
	} else {
		var len = i + flag;
		for (;i < len; i++){
			putArr.push(""+arr[i]);
		}
		i--;
	}

}
console.log(putArr);
```
